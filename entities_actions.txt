user{
    id: auto
    name: required
    date_of_birth: required
    mail: required, no duplicate
    attendance: optional
}

actions on user{
    get: retrieve users by name, retrieve users by id
    post: add new user
    put: update themselves
    patch:
    delete: delete themselves
}

event{
    id: auto
    name: required
    start_date: required
    age_range: required
    duration: optional, default 1 day
    street_name: required
    street_number: required
    city: required
    country: required
    }

actions on event{
    get: retrieve events by name, retrieve events by age range
    post: add a new event
    put: update an existing event
    patch:
    delete: delete an event
}
